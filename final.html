import numpy as np

def weighted_average(arr, weights, axis):
    return np.average(arr, axis=axis, weights=weights)

# Example usage:
arr = np.array([1, 2, 3, 4, 5])
weights = np.array([0.1, 0.2, 0.3, 0.2, 0.2])
axis = 0

result = weighted_average(arr, weights, axis)
